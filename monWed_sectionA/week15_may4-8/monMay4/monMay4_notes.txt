private static List<int> GetNumbers(string line)
{
    char[] delim = {' '};
    string[] pieces = line.Split(delim, StringSplitOptions.RemoveEmptyEntries);

    //if line was "12 -4 5"
    //What is in pieces now?
    //pieces is: {"12", "-4", "5"}


    //What if we wanted a List<int>?

    List<int> numbers = new List<int>();

    for (int i = 0; i < pieces.Length; i++) {
        //put this line in a try/catch
        int cur = Convert.ToInt32(pieces[i]);
        
        numbers.Add(cur);
    }

    return numbers;

}
