continue - skips rest of the current iteration,
    starts on the next one (after checking condition)


//What does this do?

int x = 0;
int y = 0;
do {
    x++;
    if (x == 1) {
        continue;
    }
    y+=x;
} while (x < 5);
System.out.println(y);
//would skip adding the 1
